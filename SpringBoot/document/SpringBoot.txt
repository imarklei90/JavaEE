1. SpringBoot注解

	@Controller
	@RequestMapping
	@GetMapping
	@ResponseBody

	@Value : 将属性值直接注入到Bean中
	@ConfigurationProperties
	@EnableConfigurationProperties
	@ComponentScan : 组件扫描，可以查找到Bean
	@EnableAutoConfiguration
	@Configuration
	@SpringBootApplication : 相当于@Configuration、@EnableAutoConfiguration @ComponentScan
	@RestController
	@Bean

	@ImportResource

	@EnableCaching : 开启缓存
